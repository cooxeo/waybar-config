# Waybar Configuration
# Minimal Waybar setup for Hyprland on Arch Linux
# Clean, fast, and easy to customize

# Requirements
# waybar
# hyprland
# ttf-font-awesome
# playerctl (optional)
# brightnessctl (optional)
# networkmanager (optional)

# Installation
# git clone https://github.com/cooxeo/.config.git
# mv ~/.config/waybar ~/.config/waybar-backup
# cp -r waybar-config/waybar ~/.config/


# Reload Waybar
# pkill waybar && waybar &
# hyprctl reload

# Customization Paths
# ~/.config/waybar
# ~/.config/waybar/style.css
